void fill_file(char file_name[]) {
    ofstream file(file_name, ios::out | ios::trunc);
    char buff[_MAX_PATH];
    cin.getline(buff, _MAX_PATH);
    string str = buff;
    while (str!="end") {
        str += "\n";
        file << str.c_str();
        cin.getline(buff, _MAX_PATH);
        str = buff;
    }
    file.close();
}
